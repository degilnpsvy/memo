Write a GNU asm to caltulate 2^3 + 5^2

 

# power.s

# 2^3 + 5^2

 

        .section .data

 

        .section .text

        .globl _start

 

_start:

        push $3

        push $2

        call power

        add $16, %rsp

       

        push %rax

 

        push $2

        push $5

        call power

        add $16, %rsp

 

        pop %rbx

       

        add %rax, %rbx

        mov $1, %rax

        int $0x80

 

        .type power, @function

power:

        push %rbp

        mov %rsp, %rbp

        sub $8, %rsp

 

        mov 16(%rbp), %rbx

        mov 24(%rbp), %rcx

 

        mov %rbx, -8(%rbp)

 

power_loop_start:

        cmp $1, %rcx

        je end_power

        mov -8(%rbp), %rax

        imul %rbx, %rax

        mov %rax, -8(%rbp)

        dec %rcx

        jmp power_loop_start

 

end_power:

        mov -8(%rbp), %rax

        mov %rbp, %rsp

        pop %rbp

        ret
