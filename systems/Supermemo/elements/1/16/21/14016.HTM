ifeq ($(foo),). For example, 
          bar =
          foo = $(bar)
          ifdef foo
          frobozz = yes
          else
          frobozz = no
          endif

sets ‘frobozz’ to ‘yes’, while: 

          foo =
          ifdef foo
          frobozz = yes
          else
          frobozz = no
          endif
