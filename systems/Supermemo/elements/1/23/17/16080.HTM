Example program

Listing 30-2 is a modified version of the program in Listing 30-1. It uses a mutex to protect access to the global variable glob. When we run this program with a similar command line to that used earlier, we see that glob is always reliably incremented:

$ ./thread_incr_mutex 10000000

glob = 20000000

Listing 30-2: Using a mutex to protect access to a global variable

–––––––––––––––––––––––––––––––––––––––––––––––– threads/thread_incr_mutex.c

1#include <pthread.h>

 2#include "tlpi_hdr.h"

 3static int glob = 0;

 4static pthread_mutex_t mtx = PTHREAD_MUTEX_INITIALIZER;

 5static void * /* Loop 'arg' times incrementing 'glob' */

 6threadFunc(void *arg)

 7{

 8  int loops = *((int *) arg);

 9  int loc, j, s;

10  for (j = 0; j < loops; j++) {

11    s = pthread_mutex_lock(&mtx);

12    if (s != 0)

13      errExitEN(s, "pthread_mutex_lock");

14    loc = glob;

15    loc++;

16    glob = loc;

17    s = pthread_mutex_unlock(&mtx);

18    if (s != 0)

19      errExitEN(s, "pthread_mutex_unlock");

20  }

21  return NULL;

22}

23int

24main(int argc, char *argv[])

25{

26  pthread_t t1, t2;

27  int loops, s;

28  loops = (argc > 1) ? getInt(argv[1], GN_GT_0, "num-loops") : 10000000;

29  s = pthread_create(&t1, NULL, threadFunc, &loops);

30  if (s != 0)

31    errExitEN(s, "pthread_create");

32  s = pthread_create(&t2, NULL, threadFunc, &loops);

33  if (s != 0)

34    errExitEN(s, "pthread_create");

35  s = pthread_join(t1, NULL);

36  if (s != 0)

37    errExitEN(s, "pthread_join");

38  s = pthread_join(t2, NULL);

39  if (s != 0)

40    errExitEN(s, "pthread_join");

41  printf("glob = %d\n", glob);

42  exit(EXIT_SUCCESS);

43}
