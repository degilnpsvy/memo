10.1 TLM-1 core interfaces

The transport method with the signature transport( const REQ& , RSP& ) was not part of TLM-1, but has been added in TLM-2.0.

 3  // Bidirectional blocking interfaces

 4  template < typename REQ , typename RSP > class tlm_transport_if : public virtual

 5    sc_core::sc_interface

 6  {

 7    public:

 8      virtual RSP transport(const REQ &) = 0;

 9      virtual void transport(const REQ &req , RSP &rsp)

10      {

11        rsp = transport(req);

12      }

13  };
