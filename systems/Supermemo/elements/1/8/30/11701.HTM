8.1.3 Rules

a) The default constructor tlm_phase shall set the value of the phase to 0, corresponding to the enumeration literal UNINITIALIZED_PHASE.

b) The methods tlm_phase( unsigned int), operator= and operator unsigned int shall get or set the value of the phase using the corresponding unsigned int or enum.

c) The function operator<< shall write a character string corresponding to the name of the phase to the given output stream. For example ―BEGIN_REQ‖.

d) The macro DECLARE_EXTENDED_PHASE(name_arg) shall create a new singleton class named tlm_phase_name_arg, derived from tlm_phase, and having a public method get_phase that returns a reference to the static object so created. The macro argument shall be used as the character string written by operator<< to denote the corresponding phase.

e) The intent is that the object denoted by the static const name_arg represents the extended phase that may be passed as a phase argument to nb_transport.

Example

1DECLARE_EXTENDED_PHASE(ignore_me); // Declare two extended phases

 2DECLARE_EXTENDED_PHASE(internal_ph); // Only used within target

 3struct Initiator: sc_module

 4{

 5  ...

 6  {

 7    ...

 8    phase = tlm::BEGIN_REQ;

 9    delay = sc_time(10, SC_NS);

10    socket->nb_transport_fw(trans, phase, delay);   // Send phase BEGIN_REQ to target

11    phase = ignore_me; // Set phase variable to the extended phase

12    delay = sc_time(12, SC_NS);

13    socket->nb_transport_fw(trans, phase, delay);   // Send the extended phase 2ns later

14    ...

15  }

16}

17struct Target: sc_module

18{

19  ...

20  SC_CTOR(Target) : m_peq("m_peq", this,

21                            &Target::peq_cb) {} // Register callback with PEQ

22  virtual tlm::tlm_sync_enum nb_transport_fw(tlm::tlm_generic_payload &trans,

23      tlm::tlm_phase &phase, sc_time &delay)

24  {

25    cout << "Phase = " << phase <<

26         endl; // use overloaded operator<< to print phase

27    m_peq.notify(trans, phase, delay); // Move transaction to internal queue

28    return tlm::TLM_ACCEPTED;

29  } void peq_cb(tlm::tlm_generic_payload &trans,

30                const tlm::tlm_phase &phase)   // PEQ callback

31  {

32    sc_time delay;

33    tlm::tlm_phase phase_out;

34    if (phase == tlm::BEGIN_REQ)   // Received BEGIN_REQ from initiator

35      {

36        phase_out = tlm::END_REQ;

37        delay = sc_time(10, SC_NS);

38        socket->nb_transport_bw(trans, phase_out,

39                                delay); // Send END_REQ back to initiator

40        phase_out = internal_ph; // Use extended phase to signal internal event

41        delay = sc_time(15, SC_NS);

42        m_peq.notify(trans, phase_out, delay); // Put internal event into PEQ

43      }

44    else if (phase == internal_ph)   // Received internal event

45      {

46        phase_out = tlm::BEGIN_RESP;

47        delay = sc_time(10, SC_NS);

48        socket->nb_transport_bw(trans, phase_out,

49                                delay); // Send BEGIN_RESP back to initiator

50      }

51  } // Ignore phase ignore_me from initiator

52  tlm_utils::peq_with_cb_and_phase<Target, tlm::tlm_base_protocol_types> m_peq;

53};
