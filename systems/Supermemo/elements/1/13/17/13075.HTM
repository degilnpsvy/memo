8.5. While Loops

HLL:

	// Compute n!
	c = n;
	fact = 1;
	while ( c > 1 )
	    fact *= c--;

MAL:

		# c = n
		lw      $t0, n
		sw      $t0, c
		
		# fact = 1
		li      $t1, 1
		sw      $t1, fact
	
		# while ( c > 1 )
	next_fact:
		ble     $t0, 1, fact_done
		mul     $t1, $t1, $t0
		addi    $t0, $t0, -1
		sw      $t0, c
		j       next_fact
	
	fact_done:
