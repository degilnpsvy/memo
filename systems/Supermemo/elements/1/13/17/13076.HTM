8.6. Do-While (Repeat) Loops

HLL:

	// Compute n!
	c = n;
	fact = 1;
	do
	    fact *= c--;
	while ( c > 1 );

MAL:

		# c = n
		lw      $t0, n
		sw      $t0, c
		
		# fact = 1
		li      $t1, 1
		sw      $t1, fact
	
		# Avoid multiplying by 0 or 1 if n < 2
		# Loop multiplies before checking condition.
		blt     $t0, 2, fact_done
		
		# do while
	next_fact:
		mul     $t1, $t1, $t0
		addi    $t0, $t0, -1
		sw      $t0, c
		bgt     $t0, 1, next_fact
	fact_done:
