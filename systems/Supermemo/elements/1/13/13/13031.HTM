    bit = binary digit
    nybble = 4 bits

    kibi- = 2^10
    
    mega- = 10^6
    mebi- = 2^20
    
    giga- = 10^9
    gibi- = 2^30
    
    tera- = 10^12
    tebi- = 2^40
    
    peta- = 10^15
    pebi- = 2^50
    
    exa- = 10^18
    exbi- = 2^60
    
    zeta- = 10^21
    zebi- = 2^70
