E3.1.210 SP_Process

// SP_Process - assignment form

// ============================

SP_Process = bits(32) value

if IsSecure() then

SP_Process_Secure = value;

else

SP_Process_NonSecure = value;

// SP_Process - non-assignment form

// ================================

bits(32) SP_Process

value = if IsSecure()

then SP_Process_Secure else SP_Process_NonSecure;

return value;
