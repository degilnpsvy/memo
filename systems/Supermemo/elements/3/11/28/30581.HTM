E3.1.203 S

// S[] - assignment form

// =====================

S[integer n] = bits(32) value

assert n >= 0 && n <= 31;

if (n MOD 2) == 0 then

D[n DIV 2]<31:0> = value;

else

D[n DIV 2]<63:32> = value;

return;

// S[] - non-assignment form

// =========================

bits(32) S[integer n]

assert n >= 0 && n <= 31;

if (n MOD 2) == 0 then

result = D[n DIV 2]<31:0>;

else

result = D[n DIV 2]<63:32>;

return result;
