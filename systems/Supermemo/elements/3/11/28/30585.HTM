E3.1.207 SP_Main

// SP_Main - assignment form

// =========================

SP_Main = bits(32) value

if IsSecure() then

SP_Main_Secure = value;

else

SP_Main_NonSecure = value;

// SP_Main - non-assignment form

// =============================

bits(32) SP_Main

value = if IsSecure() then SP_Main_Secure else SP_Main_NonSecure;

return value;
