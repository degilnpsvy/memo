E3.1.147 IsExceptionTargetConfigurable

// IsExceptionTargetConfigurable()

// ===============================

boolean IsExceptionTargetConfigurable(integer e)

if HaveSecurityExt() then

case e of

when NMI

configurable = TRUE;

when BusFault

configurable = TRUE;

when DebugMonitor

configurable = TRUE;

when SysTick

// If there is only 1 SysTick instance then the target domain is

// configurable.

configurable = HaveSysTick() == 1;

otherwise

// Interrupts are configurable, exceptions less than 16 that

// aren't listed above aren't.

configurable = e >= 16;

else

configurable = FALSE;

return configurable;
