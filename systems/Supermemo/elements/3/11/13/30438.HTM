E3.1.60 DefaultPermissions

// DefaultPermissions()

// ====================

Permissions DefaultPermissions(bits(32) address)

Permissions perms;

perms.ap = '01';

perms.apValid = TRUE;

perms.region = Zeros(8);

perms.regionValid = FALSE;

case address<31:29> of

when '000'

perms.xn = '0';

when '001'

perms.xn = '0';

when '010'

perms.xn = '1';

when '011'

perms.xn = '0';

when '100'

perms.xn = '0';

when '101'

perms.xn = '1';

when '110'

perms.xn = '1';

when '111'

perms.xn = '1';

return perms;
