E3.1.45 D

// D[] - assignment form

// =====================

D[integer n] = bits(64) value

assert n >= 0 && n <= 31;

if n >= 16 && VFPSmallRegisterBank() then UNDEFINED;

_D[n] = value;

return;

// D[] - non-assignment form

// =========================

bits(64) D[integer n]

assert n >= 0 && n <= 31;

if n >= 16 && VFPSmallRegisterBank() then UNDEFINED;

return _D[n];
