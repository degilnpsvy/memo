static int device_add_attrs([...])
{
 int error = 0;
 int i;

 if (!bus->dev_attrs)
  return 0;

 for (i = 0; bus->dev_attrs[i].attr.name; i++) {
  error = device_create_file(dev, &bus->dev_attrs[i]);
  if (error) {
   while (--i >= 0)
    device_remove_file(dev, &bus->dev_attrs[i]);
   break;
  }
 }
 return error;
}
