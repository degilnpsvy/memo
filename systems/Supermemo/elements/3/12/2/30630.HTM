E3.1.252 VFPExpandImm

// VFPExpandImm()

// ==============

bits(N) VFPExpandImm(bits(8) imm8, integer N)

assert N IN {32,64};

if N == 32 then E = 8; else E = 11;

constant integer F = N - E - 1;

sign = imm8<7>;

exp = NOT(imm8<6>):Replicate(imm8<6>,E-3);

frac = imm8<5:0>:Zeros(F-4);

return sign : exp : frac;
