#include <glib.h>
#include <gmodule.h>

int print(int a)
{
    g_print("%d\n",a);
}

int main()
{
    int (*print_fn_p)(int);
    GModule * self=g_module_open(NULL, G_MODULE_BIND_LAZY);
    if (g_module_symbol(self, "print", (gpointer *)&print_fn_p))
    {
        g_print("Found!\n");
        print_fn_p(3);
    }
   
    return 0;
}
