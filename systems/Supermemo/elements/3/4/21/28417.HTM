#include <stdio.h>

void * adder(int number)
{
    int add(int value){return value+number;}
    return add;
}

int main()
{
    int (*add10)(int) = adder(10);
    printf("%p, %d\n", add10, add10(1));
    int (*add100)(int) = adder(100);
    printf("%p, %d\n", add100, add100(2));
    return 0;
}

/* $ gcc -o main main.c && ./main */
