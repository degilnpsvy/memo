899/**
900 * do_gettimeofday - Returns the time of day in a timeval
901 * @tv:		pointer to the timeval to be set
902 *
903 * NOTE: Users should be converted to using getnstimeofday()
904 */
905void do_gettimeofday(struct timeval *tv)
906{
907	struct timespec64 now;
908
909	getnstimeofday64(&now);
910	tv->tv_sec = now.tv_sec;
911	tv->tv_usec = now.tv_nsec/1000;
912}
913EXPORT_SYMBOL(do_gettimeofday);
