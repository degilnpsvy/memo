#define __ATTRIBUTE_GROUPS(_name)    \
static const struct attribute_group *_name##_groups[] = { \
 &_name##_group,      \
 NULL,       \
}

#define ATTRIBUTE_GROUPS(_name)     \
static const struct attribute_group _name##_group = {  \
 .attrs = _name##_attrs,     \
};        \
__ATTRIBUTE_GROUPS(_name)
