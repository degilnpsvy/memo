bool init()

{

    //Initialization flag

    bool success = true;

 

    //Initialize SDL

    if( SDL_Init( SDL_INIT_VIDEO ) < 0 )

    {

        printf( "SDL could not initialize! SDL_Error: %s\n", SDL_GetError() );

        success = false;

    }

    else

    {

     }

 

    return success;

}
