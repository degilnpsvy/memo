223 /*
224  * Frame buffer operations
225  *
226  * LOCKING NOTE: those functions must _ALL_ be called with the console
227  * semaphore held, this is the only suitable locking mechanism we have
228  * in 2.6. Some may be called at interrupt time at this point though.
229  *
230  * The exception to this is the debug related hooks.  Putting the fb
231  * into a debug state (e.g. flipping to the kernel console) and restoring
232  * it must be done in a lock-free manner, so low level drivers should
233  * keep track of the initial console (if applicable) and may need to
234  * perform direct, unlocked hardware writes in these hooks.
235  */

419 /* A driver may set this flag to indicate that it does want a set_par to be
420  * called every time when fbcon_switch is executed. The advantage is that with
421  * this flag set you can really be sure that set_par is always called before
422  * any of the functions dependent on the correct hardware state or altering
423  * that state, even if you are using some broken X releases. The disadvantage
424  * is that it introduces unwanted delays to every console switch if set_par
425  * is slow. It is a good idea to try this flag in the drivers initialization
426  * code whenever there is a bug report related to switching between X and the
427  * framebuffer console.
428  */

509 static inline struct apertures_struct *alloc_apertures(unsigned int max_num) {
510         struct apertures_struct *a = kzalloc(sizeof(struct apertures_struct)
511                         + max_num * sizeof(struct aperture), GFP_KERNEL);
512         if (!a)
513                 return NULL;
514         a->count = max_num;
515         return a;
516 }

642 static inline void __fb_pad_aligned_buffer(u8 *dst, u32 d_pitch,
643                                            u8 *src, u32 s_pitch, u32 height)
644 {
645         u32 i, j;
646 
647         d_pitch -= s_pitch;
648 
649         for (i = height; i--; ) {
650                 /* s_pitch is a few bytes at the most, memcpy is suboptimal */
651                 for (j = 0; j < s_pitch; j++)
652                         *dst++ = *src++;
653                 dst += d_pitch;
654         }
655 }
