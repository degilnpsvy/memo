WinRT: fixed multi-touch bug whereby 2nd, 3rd, etc. fingers weren't moving

Only the first-pressed finger wpuld get reported as having moved (via SDL's
touch APIs). Subsequently pressed fingers wouldn't report as being moved, even
though the OS was reporting (to SDL) that they had moved.

    44 /*

    45 These are macros and not first class functions so that the debugger breaks

    46 on the assertion line and not in some random guts of SDL, and so each

    47 assert can have unique static variables associated with it.

    48 */

    49 

    50 #if defined(_MSC_VER)

    51 /* Don't include intrin.h here because it contains C++ code */

    52     extern void __cdecl __debugbreak(void);

    53     #define SDL_TriggerBreakpoint() __debugbreak()

    54 #elif (!defined(__NACL__) && defined(__GNUC__) && (defined(__i386__) || defined(__x86_64__)))

    55     #define SDL_TriggerBreakpoint() __asm__ __volatile__ ( "int $3\n\t" )

    56 #elif defined(HAVE_SIGNAL_H)

    57     #include <signal.h>

    58     #define SDL_TriggerBreakpoint() raise(SIGTRAP)

    59 #else

    60     /* How do we trigger breakpoints on this platform? */

    61     #define SDL_TriggerBreakpoint()

    62 #endif

    64 #if defined(__STDC_VERSION__) && (__STDC_VERSION__ >= 199901L) /* C99 supports __func__ as a standard. */

    65 #   define SDL_FUNCTION __func__

    66 #elif ((__GNUC__ >= 2) || defined(_MSC_VER))

    67 #   define SDL_FUNCTION __FUNCTION__

    68 #else

    69 #   define SDL_FUNCTION "???"

    70 #endif

    71 #define SDL_FILE    __FILE__

    72 #define SDL_LINE    __LINE__

    74 /*

    75 sizeof (x) makes the compiler still parse the expression even without

    76 assertions enabled, so the code is always checked at compile time, but

    77 doesn't actually generate code for it, so there are no side effects or

    78 expensive checks at run time, just the constant size of what x WOULD be,

    79 which presumably gets optimized out as unused.

    80 This also solves the problem of...

    81 

    82     int somevalue = blah();

    83     SDL_assert(somevalue == 1);

    84 

    85 ...which would cause compiles to complain that somevalue is unused if we

    86 disable assertions.

    87 */

    95 #define SDL_disabled_assert(condition) \

    96     do { (void) sizeof ((condition)); } while (SDL_NULL_WHILE_LOOP_CONDITION)

    98 typedef enum

    99 {

   100     SDL_ASSERTION_RETRY,  /**< Retry the assert immediately. */

   101     SDL_ASSERTION_BREAK,  /**< Make the debugger trigger a breakpoint. */

   102     SDL_ASSERTION_ABORT,  /**< Terminate the program. */

   103     SDL_ASSERTION_IGNORE,  /**< Ignore the assert. */

   104     SDL_ASSERTION_ALWAYS_IGNORE  /**< Ignore the assert from now on. */

   105 } SDL_assert_state;

   106 

   107 typedef struct SDL_assert_data

   108 {

   109     int always_ignore;

   110     unsigned int trigger_count;

   111     const char *condition;

   112     const char *filename;

   113     int linenum;

   114     const char *function;

   115     const struct SDL_assert_data *next;

   116 } SDL_assert_data;

   118 #if (SDL_ASSERT_LEVEL > 0)

   119 

   120 /* Never call this directly. Use the SDL_assert* macros. */

   121 extern DECLSPEC SDL_assert_state SDLCALL SDL_ReportAssertion(SDL_assert_data *,

   122                                                              const char *,

   123                                                              const char *, int)

   124 #if defined(__clang__)

   125 #if __has_feature(attribute_analyzer_noreturn)

   126 /* this tells Clang's static analysis that we're a custom assert function,

   127    and that the analyzer should assume the condition was always true past this

   128    SDL_assert test. */

   129    __attribute__((analyzer_noreturn))

   130 #endif

   131 #endif

   132 ;

   133 

   134 /* the do {} while(0) avoids dangling else problems:

   135     if (x) SDL_assert(y); else blah();

   136        ... without the do/while, the "else" could attach to this macro's "if".

   137    We try to handle just the minimum we need here in a macro...the loop,

   138    the static vars, and break points. The heavy lifting is handled in

   139    SDL_ReportAssertion(), in SDL_assert.c.

   140 */

   141 #define SDL_enabled_assert(condition) \

   142     do { \

   143         while ( !(condition) ) { \

   144             static struct SDL_assert_data sdl_assert_data = { \

   145                 0, 0, #condition, 0, 0, 0, 0 \

   146             }; \

   147             const SDL_assert_state sdl_assert_state = SDL_ReportAssertion(&sdl_assert_data, SDL_FUNCTION, SDL_FILE, SDL_LINE); \

   148             if (sdl_assert_state == SDL_ASSERTION_RETRY) { \

   149                 continue; /* go again. */ \

   150             } else if (sdl_assert_state == SDL_ASSERTION_BREAK) { \

   151                 SDL_TriggerBreakpoint(); \

   152             } \

   153             break; /* not retrying. */ \

   154         } \

   155     } while (SDL_NULL_WHILE_LOOP_CONDITION)

   156 

   157 #endif  /* enabled assertions support code */

   158 

   159 /* Enable various levels of assertions. */

   160 #if SDL_ASSERT_LEVEL == 0   /* assertions disabled */

   161 #   define SDL_assert(condition) SDL_disabled_assert(condition)

   162 #   define SDL_assert_release(condition) SDL_disabled_assert(condition)

   163 #   define SDL_assert_paranoid(condition) SDL_disabled_assert(condition)

   164 #elif SDL_ASSERT_LEVEL == 1  /* release settings. */

   165 #   define SDL_assert(condition) SDL_disabled_assert(condition)

   166 #   define SDL_assert_release(condition) SDL_enabled_assert(condition)

   167 #   define SDL_assert_paranoid(condition) SDL_disabled_assert(condition)

   168 #elif SDL_ASSERT_LEVEL == 2  /* normal settings. */

   169 #   define SDL_assert(condition) SDL_enabled_assert(condition)

   170 #   define SDL_assert_release(condition) SDL_enabled_assert(condition)

   171 #   define SDL_assert_paranoid(condition) SDL_disabled_assert(condition)

   172 #elif SDL_ASSERT_LEVEL == 3  /* paranoid settings. */

   173 #   define SDL_assert(condition) SDL_enabled_assert(condition)

   174 #   define SDL_assert_release(condition) SDL_enabled_assert(condition)

   175 #   define SDL_assert_paranoid(condition) SDL_enabled_assert(condition)

   176 #else

   177 #   error Unknown assertion level.

   178 #endif

   179 

   180 /* this assertion is never disabled at any level. */

   181 #define SDL_assert_always(condition) SDL_enabled_assert(condition)

   184 typedef SDL_assert_state (SDLCALL *SDL_AssertionHandler)(

   185                                  const SDL_assert_data* data, void* userdata);

   186 

   187 /**

   188  *  \brief Set an application-defined assertion handler.

   189  *

   190  *  This allows an app to show its own assertion UI and/or force the

   191  *  response to an assertion failure. If the app doesn't provide this, SDL

   192  *  will try to do the right thing, popping up a system-specific GUI dialog,

   193  *  and probably minimizing any fullscreen windows.

   194  *

   195  *  This callback may fire from any thread, but it runs wrapped in a mutex, so

   196  *  it will only fire from one thread at a time.

   197  *

   198  *  Setting the callback to NULL restores SDL's original internal handler.

   199  *

   200  *  This callback is NOT reset to SDL's internal handler upon SDL_Quit()!

   201  *

   202  *  \return SDL_assert_state value of how to handle the assertion failure.

   203  *

   204  *  \param handler Callback function, called when an assertion fails.

   205  *  \param userdata A pointer passed to the callback as-is.

   206  */

   207 extern DECLSPEC void SDLCALL SDL_SetAssertionHandler(

   208                                             SDL_AssertionHandler handler,

   209                                             void *userdata);

   210 

   211 /**

   212  *  \brief Get the default assertion handler.

   213  *

   214  *  This returns the function pointer that is called by default when an

   215  *   assertion is triggered. This is an internal function provided by SDL,

   216  *   that is used for assertions when SDL_SetAssertionHandler() hasn't been

   217  *   used to provide a different function.

   218  *

   219  *  \return The default SDL_AssertionHandler that is called when an assert triggers.

   220  */

   221 extern DECLSPEC SDL_AssertionHandler SDLCALL SDL_GetDefaultAssertionHandler(void);

   222 

   223 /**

   224  *  \brief Get the current assertion handler.

   225  *

   226  *  This returns the function pointer that is called when an assertion is

   227  *   triggered. This is either the value last passed to

   228  *   SDL_SetAssertionHandler(), or if no application-specified function is

   229  *   set, is equivalent to calling SDL_GetDefaultAssertionHandler().

   230  *

   231  *   \param puserdata Pointer to a void*, which will store the "userdata"

   232  *                    pointer that was passed to SDL_SetAssertionHandler().

   233  *                    This value will always be NULL for the default handler.

   234  *                    If you don't care about this data, it is safe to pass

   235  *                    a NULL pointer to this function to ignore it.

   236  *  \return The SDL_AssertionHandler that is called when an assert triggers.

   237  */

   238 extern DECLSPEC SDL_AssertionHandler SDLCALL SDL_GetAssertionHandler(void **puserdata);

   239 

   240 /**

   241  *  \brief Get a list of all assertion failures.

   242  *

   243  *  Get all assertions triggered since last call to SDL_ResetAssertionReport(),

   244  *  or the start of the program.

   245  *

   246  *  The proper way to examine this data looks something like this:

   247  *

   248  *  <code>

   249  *  const SDL_assert_data *item = SDL_GetAssertionReport();

   250  *  while (item) {

   251  *      printf("'%s', %s (%s:%d), triggered %u times, always ignore: %s.\n",

   252  *             item->condition, item->function, item->filename,

   253  *             item->linenum, item->trigger_count,

   254  *             item->always_ignore ? "yes" : "no");

   255  *      item = item->next;

   256  *  }

   257  *  </code>

   258  *

   259  *  \return List of all assertions.

   260  *  \sa SDL_ResetAssertionReport

   261  */

   262 extern DECLSPEC const SDL_assert_data * SDLCALL SDL_GetAssertionReport(void);

   263 

   264 /**

   265  *  \brief Reset the list of all assertion failures.

   266  *

   267  *  Reset list of all assertions triggered.

   268  *

   269  *  \sa SDL_GetAssertionReport

   270  */

   271 extern DECLSPEC void SDLCALL SDL_ResetAssertionReport(void);
