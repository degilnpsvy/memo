   350 /* Note that memset() is a byte assignment and this is a 32-bit assignment, so they're not directly equivalent. */

   351 SDL_FORCE_INLINE void SDL_memset4(void *dst, Uint32 val, size_t dwords)

   352 {

   353 #if defined(__GNUC__) && defined(i386)

   354     int u0, u1, u2;

   355     __asm__ __volatile__ (

   356         "cld \n\t"

   357         "rep ; stosl \n\t"

   358         : "=&D" (u0), "=&a" (u1), "=&c" (u2)

   359         : "0" (dst), "1" (val), "2" (SDL_static_cast(Uint32, dwords))

   360         : "memory"

   361     );

   362 #else

   363     size_t _n = (dwords + 3) / 4;

   364     Uint32 *_p = SDL_static_cast(Uint32 *, dst);

   365     Uint32 _val = (val);

   366     if (dwords == 0)

   367         return;

   368     switch (dwords % 4)

   369     {

   370         case 0: do {    *_p++ = _val;

   371         case 3:         *_p++ = _val;

   372         case 2:         *_p++ = _val;

   373         case 1:         *_p++ = _val;

   374         } while ( --_n );

   375     }

   376 #endif

   377 }
