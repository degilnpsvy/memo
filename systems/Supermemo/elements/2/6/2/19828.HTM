SDL_RenderCopy

Use this function to copy a portion of the texture to the current rendering target.

Syntax

切换行号显示
int SDL_RenderCopy(SDL_Renderer*   renderer,
                   SDL_Texture*    texture,
                   const SDL_Rect* srcrect,
                   const SDL_Rect* dstrect)

Function Parameters

renderer

the rendering context

texture

the source texture; see Remarks for details

srcrect

the source SDL_Rect structure or NULL for the entire texture

dstrect

the destination SDL_Rect structure or NULL for the entire rendering target; the texture will be stretched to fill the given rectangle

Return Value

Returns 0 on success or a negative error code on failure; call SDL_GetError() for more information.

   1 #include "SDL.h"
   2 #define SHAPE_SIZE 16
   3 
   4 int main(int argc, char *argv[])
   5 {
   6   SDL_Window* Main_Window;
   7   SDL_Renderer* Main_Renderer;
   8   SDL_Surface* Loading_Surf;
   9   SDL_Texture* Background_Tx;
  10   SDL_Texture* BlueShapes;
  11 
  12   /* Rectangles for drawing which will specify source (inside the texture)
  13   and target (on the screen) for rendering our textures. */
  14   SDL_Rect SrcR;
  15   SDL_Rect DestR;
  16 
  17   SrcR.x = 0;
  18   SrcR.y = 0;
  19   SrcR.w = SHAPE_SIZE;
  20   SrcR.h = SHAPE_SIZE;
  21 
  22   DestR.x = 640 / 2 - SHAPE_SIZE / 2;
  23   DestR.y = 580 / 2 - SHAPE_SIZE / 2;
  24   DestR.w = SHAPE_SIZE;
  25   DestR.h = SHAPE_SIZE;
  26 
  27 
  28   /* Before we can render anything, we need a window and a renderer */
  29   Main_Window = SDL_CreateWindow("SDL_RenderCopy Example",
  30   SDL_WINDOWPOS_UNDEFINED, SDL_WINDOWPOS_UNDEFINED, 640, 580, 0);
  31   Main_Renderer = SDL_CreateRenderer(Main_Window, -1, SDL_RENDERER_ACCELERATED);
  32 
  33   /* The loading of the background texture. Since SDL_LoadBMP returns
  34   a surface, we convert it to a texture afterwards for fast accelerated
  35   blitting. */
  36   Loading_Surf = SDL_LoadBMP("Background.bmp");
  37   Background_Tx = SDL_CreateTextureFromSurface(Main_Renderer, Loading_Surf);
  38   SDL_FreeSurface(Loading_Surf);  // we got the texture now -> free surface
  39 
  40   /* Load an additional texture */
  41   Loading_Surf = SDL_LoadBMP("Blueshapes.bmp");
  42   BlueShapes = SDL_CreateTextureFromSurface(Main_Renderer, Loading_Surf);
  43   SDL_FreeSurface(Loading_Surf);
  44 
  45   /* now onto the fun part.
  46   This will render a rotating selection of the blue shapes
  47   in the middle of the screen */
  48   int i;
  49   int n;
  50   for (i = 0; i < 2; ++i) {
  51     for(n = 0; n < 4; ++n) {
  52       SrcR.x = SHAPE_SIZE * (n % 2);
  53       if (n > 1) {
  54         SrcR.y = SHAPE_SIZE;
  55       } else {
  56         SrcR.y = 0;
  57       }
  58 
  59       /* render background, whereas NULL for source and destination
  60       rectangles just means "use the default" */
  61       SDL_RenderCopy(Main_Renderer, Background_Tx, NULL, NULL);
  62 
  63       /* render the current animation step of our shape */
  64       SDL_RenderCopy(Main_Renderer, BlueShapes, &SrcR, &DestR);
  65       SDL_RenderPresent(Main_Renderer);
  66       SDL_Delay(500);
  67     }
  68   }
  69 
  70 
  71   /* The renderer works pretty much like a big canvas:
  72   when you RenderCopy you are adding paint, each time adding it
  73   on top.
  74   You can change how it blends with the stuff that
  75   the new data goes over.
  76   When your 'picture' is complete, you show it
  77   by using SDL_RenderPresent. */
  78 
  79   /* SDL 1.2 hint: If you're stuck on the whole renderer idea coming
  80   from 1.2 surfaces and blitting, think of the renderer as your
  81   main surface, and SDL_RenderCopy as the blit function to that main
  82   surface, with SDL_RenderPresent as the old SDL_Flip function.*/
  83 
  84   SDL_DestroyTexture(BlueShapes);
  85   SDL_DestroyTexture(Background_Tx);
  86   SDL_DestroyRenderer(Main_Renderer);
  87   SDL_DestroyWindow(Main_Window);
  88   SDL_Quit();
  89 
  90 
  91   return 0;
  92 }

Remarks

The texture is blended with the destination based on its blend mode set with SDL_SetTextureBlendMode().

The texture color is affected based on its color modulation set by SDL_SetTextureColorMod().

The texture alpha is affected based on its alpha modulation set by SDL_SetTextureAlphaMod().
