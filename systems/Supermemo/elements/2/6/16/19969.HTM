SDL_GetWindowWMInfo

Use this function to get driver specific information about a window.

Syntax

切换行号显示
SDL_bool SDL_GetWindowWMInfo(SDL_Window*    window,
                             SDL_SysWMinfo* info)

Function Parameters

window

the window about which information is being requested

info

an SDL_SysWMinfo structure filled in with window information; see Remarks for details

Return Value

Returns SDL_TRUE if the function is implemented and the version member of the info struct is valid, or SDL_FALSE if the information could not be retrieved; call SDL_GetError() for more information.

Code Examples

切换行号显示
   1 #include "SDL.h"
   2 #include "SDL_syswm.h"
   3 
   4 int main(int argc, char *argv[]) {
   5   SDL_Window* window;
   6   SDL_SysWMinfo info;
   7 
   8   SDL_Init(0);
   9 
  10   window = SDL_CreateWindow("", 0, 0, 0, 0, SDL_WINDOW_HIDDEN);
  11 
  12   SDL_VERSION(&info.version); // initialize info structure with SDL version info
  13 
  14   if(SDL_GetWindowWMInfo(window,&info)) { // the call returns true on success
  15     // success
  16     const char *subsystem = "an unknown system!";
  17     switch(info.subsystem) {
  18       case SDL_SYSWM_UNKNOWN:   break;
  19       case SDL_SYSWM_WINDOWS:   subsystem = "Microsoft Windows(TM)";  break;
  20       case SDL_SYSWM_X11:       subsystem = "X Window System";        break;
  21 #if SDL_VERSION_ATLEAST(2, 0, 3)
  22       case SDL_SYSWM_WINRT:     subsystem = "WinRT";                  break;
  23 #endif
  24       case SDL_SYSWM_DIRECTFB:  subsystem = "DirectFB";               break;
  25       case SDL_SYSWM_COCOA:     subsystem = "Apple OS X";             break;
  26       case SDL_SYSWM_UIKIT:     subsystem = "UIKit";                  break;
  27 #if SDL_VERSION_ATLEAST(2, 0, 2)
  28       case SDL_SYSWM_WAYLAND:   subsystem = "Wayland";                break;
  29       case SDL_SYSWM_MIR:       subsystem = "Mir";                    break;
  30 #endif
  31 #if SDL_VERSION_ATLEAST(2, 0, 4)
  32       case SDL_SYSWM_ANDROID:   subsystem = "Android";                break;
  33 #endif
  34     }
  35 
  36     SDL_Log("This program is running SDL version %d.%d.%d on %s\n", 
  37         (int)info.version.major,
  38         (int)info.version.minor,
  39         (int)info.version.patch,
  40         subsystem);
  41   } else {
  42     // call failed
  43     SDL_LogError(SDL_LOG_CATEGORY_ERROR, "Couldn't get window information: %s\n", SDL_GetError());
  44   }
  45 
  46   SDL_DestroyWindow(window);
  47   SDL_Quit();
  48 
  49   return 0;
  50 }

Remarks

You must include SDL_syswm.h for the declaration of SDL_SysWMinfo.

The info structure must be initialized with the SDL version, and is then filled in with information about the given window, as shown in the Code Example.
