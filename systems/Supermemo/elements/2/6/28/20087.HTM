Introduction

The SDL Haptic subsystem allows you to control haptic (force feedback) devices.

The basic usage is as follows:

  1. Initialize the Subsystem (SDL_INIT_HAPTIC)
  2. Open a Haptic Device

    1. SDL_HapticOpen() to open from index

    2. SDL_HapticOpenFromJoystick() to open from an existing joystick

  3. Create an effect (SDL_HapticEffect)

  4. Upload the effect with SDL_HapticNewEffect()

  5. Run the effect with SDL_HapticRunEffect()

  6. (optional) Free the effect with SDL_HapticDestroyEffect()

  7. Close the haptic device with SDL_HapticClose()

Code Examples

Simple rumble example:

切换行号显示
   1 SDL_Haptic *haptic;
   2 
   3 // Open the device
   4 haptic = SDL_HapticOpen( 0 );
   5 if (haptic == NULL)
   6    return -1;
   7 
   8 // Initialize simple rumble
   9 if (SDL_HapticRumbleInit( haptic ) != 0)
  10    return -1;
  11 
  12 // Play effect at 50% strength for 2 seconds
  13 if (SDL_HapticRumblePlay( haptic, 0.5, 2000 ) != 0)
  14    return -1;
  15 SDL_Delay( 2000 );
  16 
  17 // Clean up
  18 SDL_HapticClose( haptic );

Complete example:

切换行号显示
   1 int test_haptic( SDL_Joystick * joystick ) {
   2  SDL_Haptic *haptic;
   3  SDL_HapticEffect effect;
   4  int effect_id;
   5 
   6  // Open the device
   7  haptic = SDL_HapticOpenFromJoystick( joystick );
   8  if (haptic == NULL) return -1; // Most likely joystick isn't haptic
   9 
  10  // See if it can do sine waves
  11  if ((SDL_HapticQuery(haptic) & SDL_HAPTIC_SINE)==0) {
  12   SDL_HapticClose(haptic); // No sine effect
  13   return -1;
  14  }
  15 
  16  // Create the effect
  17  memset( &effect, 0, sizeof(SDL_HapticEffect) ); // 0 is safe default
  18  effect.type = SDL_HAPTIC_SINE;
  19  effect.periodic.direction.type = SDL_HAPTIC_POLAR; // Polar coordinates
  20  effect.periodic.direction.dir[0] = 18000; // Force comes from south
  21  effect.periodic.period = 1000; // 1000 ms
  22  effect.periodic.magnitude = 20000; // 20000/32767 strength
  23  effect.periodic.length = 5000; // 5 seconds long
  24  effect.periodic.attack_length = 1000; // Takes 1 second to get max strength
  25  effect.periodic.fade_length = 1000; // Takes 1 second to fade away
  26 
  27  // Upload the effect
  28  effect_id = SDL_HapticNewEffect( haptic, &effect );
  29 
  30  // Test the effect
  31  SDL_HapticRunEffect( haptic, effect_id, 1 );
  32  SDL_Delay( 5000); // Wait for the effect to finish
  33 
  34  // We destroy the effect, although closing the device also does this
  35  SDL_HapticDestroyEffect( haptic, effect_id );
  36 
  37  // Close the device
  38  SDL_HapticClose(haptic);
  39 
  40  return 0; // Success
  41 }
