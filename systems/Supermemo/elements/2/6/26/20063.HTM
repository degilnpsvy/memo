    79 /**

    80  *  \brief Retrieve the current state of the mouse.

    81  *

    82  *  The current button state is returned as a button bitmask, which can

    83  *  be tested using the SDL_BUTTON(X) macros, and x and y are set to the

    84  *  mouse cursor position relative to the focus window for the currently

    85  *  selected mouse.  You can pass NULL for either x or y.

    86  */

    87 extern DECLSPEC Uint32 SDLCALL SDL_GetMouseState(int *x, int *y);

    88 

    89 /**

    90  *  \brief Get the current state of the mouse, in relation to the desktop

    91  *

    92  *  This works just like SDL_GetMouseState(), but the coordinates will be

    93  *  reported relative to the top-left of the desktop. This can be useful if

    94  *  you need to track the mouse outside of a specific window and

    95  *  SDL_CaptureMouse() doesn't fit your needs. For example, it could be

    96  *  useful if you need to track the mouse while dragging a window, where

    97  *  coordinates relative to a window might not be in sync at all times.

    98  *

    99  *  \note SDL_GetMouseState() returns the mouse position as SDL understands

   100  *        it from the last pump of the event queue. This function, however,

   101  *        queries the OS for the current mouse position, and as such, might

   102  *        be a slightly less efficient function. Unless you know what you're

   103  *        doing and have a good reason to use this function, you probably want

   104  *        SDL_GetMouseState() instead.

   105  *

   106  *  \param x Returns the current X coord, relative to the desktop. Can be NULL.

   107  *  \param y Returns the current Y coord, relative to the desktop. Can be NULL.

   108  *  \return The current button state as a bitmask, which can be tested using the SDL_BUTTON(X) macros.

   109  *

   110  *  \sa SDL_GetMouseState

   111  */

   112 extern DECLSPEC Uint32 SDLCALL SDL_GetGlobalMouseState(int *x, int *y);

   113 

   114 /**

   115  *  \brief Retrieve the relative state of the mouse.

   116  *

   117  *  The current button state is returned as a button bitmask, which can

   118  *  be tested using the SDL_BUTTON(X) macros, and x and y are set to the

   119  *  mouse deltas since the last call to SDL_GetRelativeMouseState().

   120  */

   121 extern DECLSPEC Uint32 SDLCALL SDL_GetRelativeMouseState(int *x, int *y);

   145 /**

   146  *  \brief Set relative mouse mode.

   147  *

   148  *  \param enabled Whether or not to enable relative mode

   149  *

   150  *  \return 0 on success, or -1 if relative mode is not supported.

   151  *

   152  *  While the mouse is in relative mode, the cursor is hidden, and the

   153  *  driver will try to report continuous motion in the current window.

   154  *  Only relative motion events will be delivered, the mouse position

   155  *  will not change.

   156  *

   157  *  \note This function will flush any pending mouse motion.

   158  *

   159  *  \sa SDL_GetRelativeMouseMode()

   160  */

   161 extern DECLSPEC int SDLCALL SDL_SetRelativeMouseMode(SDL_bool enabled);

   162 

   163 /**

   164  *  \brief Capture the mouse, to track input outside an SDL window.

   165  *

   166  *  \param enabled Whether or not to enable capturing

   167  *

   168  *  Capturing enables your app to obtain mouse events globally, instead of

   169  *  just within your window. Not all video targets support this function.

   170  *  When capturing is enabled, the current window will get all mouse events,

   171  *  but unlike relative mode, no change is made to the cursor and it is

   172  *  not restrained to your window.

   173  *

   174  *  This function may also deny mouse input to other windows--both those in

   175  *  your application and others on the system--so you should use this

   176  *  function sparingly, and in small bursts. For example, you might want to

   177  *  track the mouse while the user is dragging something, until the user

   178  *  releases a mouse button. It is not recommended that you capture the mouse

   179  *  for long periods of time, such as the entire time your app is running.

   180  *

   181  *  While captured, mouse events still report coordinates relative to the

   182  *  current (foreground) window, but those coordinates may be outside the

   183  *  bounds of the window (including negative values). Capturing is only

   184  *  allowed for the foreground window. If the window loses focus while

   185  *  capturing, the capture will be disabled automatically.

   186  *

   187  *  While capturing is enabled, the current window will have the

   188  *  SDL_WINDOW_MOUSE_CAPTURE flag set.

   189  *

   190  *  \return 0 on success, or -1 if not supported.

   191  */

   192 extern DECLSPEC int SDLCALL SDL_CaptureMouse(SDL_bool enabled);

   201 /**

   202  *  \brief Create a cursor, using the specified bitmap data and

   203  *         mask (in MSB format).

   204  *

   205  *  The cursor width must be a multiple of 8 bits.

   206  *

   207  *  The cursor is created in black and white according to the following:

   208  *  <table>

   209  *  <tr><td> data </td><td> mask </td><td> resulting pixel on screen </td></tr>

   210  *  <tr><td>  0   </td><td>  1   </td><td> White </td></tr>

   211  *  <tr><td>  1   </td><td>  1   </td><td> Black </td></tr>

   212  *  <tr><td>  0   </td><td>  0   </td><td> Transparent </td></tr>

   213  *  <tr><td>  1   </td><td>  0   </td><td> Inverted color if possible, black

   214  *                                         if not. </td></tr>

   215  *  </table>

   216  *

   217  *  \sa SDL_FreeCursor()

   218  */

   219 extern DECLSPEC SDL_Cursor *SDLCALL SDL_CreateCursor(const Uint8 * data,

   220                                                      const Uint8 * mask,

   221                                                      int w, int h, int hot_x,

   222                                                      int hot_y);

   272 /**

   273  *  Used as a mask when testing buttons in buttonstate.

   274  *   - Button 1:  Left mouse button

   275  *   - Button 2:  Middle mouse button

   276  *   - Button 3:  Right mouse button

   279 #define SDL_BUTTON_LEFT     1

   280 #define SDL_BUTTON_MIDDLE   2

   281 #define SDL_BUTTON_RIGHT    3

   282 #define SDL_BUTTON_X1       4

   283 #define SDL_BUTTON_X2       5

   284 #define SDL_BUTTON_LMASK    SDL_BUTTON(SDL_BUTTON_LEFT)

   285 #define SDL_BUTTON_MMASK    SDL_BUTTON(SDL_BUTTON_MIDDLE)

   286 #define SDL_BUTTON_RMASK    SDL_BUTTON(SDL_BUTTON_RIGHT)
