666struct pci_driver {
667	struct list_head node;

674	int  (*resume_early) (struct pci_dev *dev);
675	int  (*resume) (struct pci_dev *dev);	                /* Device woken up */

677	int (*sriov_configure) (struct pci_dev *dev, int num_vfs); /* PF pdev */
