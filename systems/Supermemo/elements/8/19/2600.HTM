Realsil Coding style

switch-case statement:

switch

  {

    case 1:

      printf("this is case 1\n");

      break;

    case 2:

      printf("this is case 2\n");

      break;

    case 3:

      {

        printf("this is case 3\n");

        printf("and it has its own scope\n");

      }

      break;

  }
