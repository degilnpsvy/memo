. Capitalize the first letter of the words in a symbol.

class MyClass

{

  public:

    MyClass();

    ~MyClass();

    int hehe;

  private:

    int haha;

}

. Use underscore to connect words in a symbol.

class my_class

{

  public:

    my_class();

    ~my_class();

    int hehe;

  private:

    int haha;

}
